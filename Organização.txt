void criar_tabela() {
    char name[30];
    printf("Qual será o nome de sua tabela? ");
    fgets(name, sizeof(name), stdin);

    // Remova o caractere de nova linha (\n) do final da string
    size_t length = strlen(name);
    if (length > 0 && name[length - 1] == '\n') {
        name[length - 1] = '\0';
    }

    // Crie o caminho do arquivo concatenando as strings
    char nome_arquivo[100] = "arquivo/";
    strcat(nome_arquivo, name);
    strcat(nome_arquivo, ".txt");

    FILE *arquivo;
    // Abre o arquivo para escrita (append)
    arquivo = fopen(nome_arquivo, "a");

    // Verifica se o arquivo foi aberto com sucesso
    if (arquivo == NULL) {
        fprintf(stderr, "Erro ao abrir o arquivo.\n");
        return;
    }

    // Escreve no arquivo
    fprintf(arquivo, "Conteúdo da tabela %s\n", name);

    // Fecha o arquivo
    fclose(arquivo);
}

int main() {
    criar_tabela();
    return 0;
}
